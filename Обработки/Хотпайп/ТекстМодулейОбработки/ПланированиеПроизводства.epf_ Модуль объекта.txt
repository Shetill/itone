Функция ПолучитьДанныеРегистрации()
	
	ДанныеРегистрации = Новый Структура;
	ДанныеРегистрации.Вставить("Наименование", 		"Планирование производства");
	ДанныеРегистрации.Вставить("БезопасныйРежим", 	Ложь);
	ДанныеРегистрации.Вставить("Версия",			"1.0.0");
	ДанныеРегистрации.Вставить("Информация",		"Планирование производства (внешняя обработка)");
	ДанныеРегистрации.Вставить("Команды",			ТаблицаКоманд());
	ДанныеРегистрации.Вставить("Вид",				"ДополнительнаяОбработка");
	
	Возврат ДанныеРегистрации;
	 
КонецФункции

Функция ТаблицаКоманд()

	тзКоманды.Колонки.Добавить("Представление");

	тзКоманды.Колонки.Добавить("Модификатор");

	Возврат тзКоманды;
